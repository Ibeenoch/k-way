import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

async function searchWordInRelatedTable(word: string) {
  try {
    // Fetch all records matching the search criteria
    const results = await prisma.relatedTable.findMany({
      where: {
        fieldName: {
          contains: word,
        },
      },
    });

    // Shuffle the array of results
    const shuffledResults = shuffleArray(results);

    console.log('Shuffled Results:', shuffledResults);
  } catch (error) {
    console.error('Error searching word in related table:', error);
  } finally {
    await prisma.$disconnect();
  }
}

// Utility function to shuffle an array
function shuffleArray<T>(array: T[]): T[] {
  const shuffledArray = [...array];
  for (let i = shuffledArray.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [shuffledArray[i], shuffledArray[j]] = [shuffledArray[j], shuffledArray[i]];
  }
  return shuffledArray;
}

// Call the function with the word you want to search for
searchWordInRelatedTable('your_word');
